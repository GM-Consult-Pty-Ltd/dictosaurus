@override
Map<PartOfSpeech, Set<String>> antonymsMap() {
  final Map<PartOfSpeech, Set<String>> retVal = {};
  for (final e in variants) {
    final value = retVal[e.partOfSpeech] ?? {};
    value.addAll(e.antonyms);
    retVal[e.partOfSpeech] = value;
  }
  return retVal;
}
